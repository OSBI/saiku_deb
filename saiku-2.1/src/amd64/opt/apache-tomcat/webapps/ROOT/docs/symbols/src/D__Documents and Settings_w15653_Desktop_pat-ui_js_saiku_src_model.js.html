 25     /**
 26      * Handle all AJAX requests.
 27      * @param paramters {Object} Parameters for AJAX requests.
 28      */
 29     request : function(parameters) {
 30         if (typeof parameters.method == "undefined")
 31             parameters.method = "GET";
 32         if (typeof parameters.data == "undefined")
 33             parameters.data = {};
 34         if (typeof parameters.success == "undefined")
 35             parameters.success = function() {};
 36         if (typeof parameters.error == "undefined")
 37             parameters.error = model.server_error;
 38         
 39         $.ajax({
 40             type: parameters.method,
 41             url: "/pat-ui/fixtures/rest/" + parameters.url,
 42             //url: "rest/" + parameters.url, CHANGE
 43             dataType: 'json',
 44             username: model.username,
 45             password: model.password,
 46             success: parameters.success,
 47             error: parameters.error
 48         });
 49     },

 91     /** Get the sessionid and based on the username and unhide the UI. */
 92     get_session : function() {
 93         model.request({
 94             method: "POST",
 95             url: model.username + "/session",
 96             success: function(data, textStatus, XMLHttpRequest) {
 97                 model.session_id = data['@sessionid'];
 98                 model.connections = data;
 99                 view.draw_ui();
100                 controller.add_tab();
101             }
102         });

105     /**
106  * Populate the dimension and measure tree and initialise draggable,
107  * droppable and sortable items.
108  * @param tab_index {Integer} Index of the selected tab.
109  */
110     new_query : function(tab_index) {
111 
112         /** Find the selected cube. */
113         $cube = view.tabs.tabs[tab_index].content.find(".cubes option:selected");
114 
115         /** Check if the cube is valid if so then display an error. */
116         data = view.tabs.tabs[tab_index].data['navigation'][$cube.attr('value')];
117         if (typeof data == "undefined") {
118             view.show_dialog('Error', 'There was an error loading that cube.<br/>Please close the tab and try again.', 'error')
119             return;
120         }
121 
122         // Reference for the selected tabs content.
123         $tab = view.tabs.tabs[tab_index].content;
124 
125         // Get a list of available dimensions and measures.
126         model.request({
127             method : "POST",
128             url : model.username + "/query/" + data['schema'] + "/" + data['cube'] + "/newquery",
129             success : function(data, textStatus, XMLHttpRequest) {
130                 /** Load dimensions into a tree. */
131                 view.load_dimensions($tab, data.axis.dimensions);
132                 /** Load measures into a tree. */
133                 view.load_measures($tab, data.axis.dimensions);
134 
135                 /** Activate hide and show on trees. */
136                 $tab.find('.root').parent().children('ul').children('li').hide();
137                 /** When the root item is clicked show it's children. */
138                 $tab.find('.root').click(function() {
139                     $(this).parent().children("ul").children('li').toggle();
140                     if ($(this).parent().hasClass('expand')) {
141                         $(this).parent().removeClass('expand').addClass('collapsed')
142                         .find('a.folder_expand')
143                         .removeClass('folder_expand')
144                         .addClass('folder_collapsed');
145                     }else{
146                         $(this).parent().removeClass('collapsed').addClass('expand')
147                         .find('a.folder_collapsed')
148                         .removeClass('folder_collapsed')
149                         .addClass('folder_expand');
150                     }
151                     return false;
152                 });
153 
154                 /** Lookups. */
155                 $both_trees = $tab.find('.dimension_tree, .measure_tree');
156                 $both_dropzones = view.tabs.tabs[tab_index].content.find('.rows ul, .columns ul');
157                 $row_dropzone = view.tabs.tabs[tab_index].content.find('.rows ul');
158                 $column_dropzone = view.tabs.tabs[tab_index].content.find('.columns ul');
159                 $measure_tree = $tab.find('.measure_tree li ul li');
160                 $dimension_tree = $tab.find('.dimension_tree li ul li');
161                 $dimensions_dropzone = view.tabs.tabs[tab_index].content.find('.dimensions_group');
162                 $measures_dropzone = view.tabs.tabs[tab_index].content.find('.measures_group');
163 
164                 /** Dimensions. */
165 
166                 /** Enable dragging on the dimension tree. */
167                 $dimension_tree.draggable({
168                     cancel : ".not-draggable",
169                     helper : "clone",
170                     opacity : 0.60,
171                     /** Connect to the dimension dropzone. */
172                     connectToSortable : $dimensions_dropzone,
173                     cursorAt : {
174                         top : 7,
175                         left : 5
176                     }
177                 }).disableSelection();
178 
179 
180                 /** Make the dimension lists sortable and connectable. */
181                 $dimensions_dropzone.sortable({
182                     cancel : 'placeholder_dimension',
183                     placeholder : 'placeholder_dimension_sort',
184                     items : 'li:not(.placeholder_dimension)',
185                     connectWith : $dimensions_dropzone,
186                     opacity: 0.60,
187                     distance: 30,
188                     cursorAt : {
189                         top : 7,
190                         left : 5
191                     },
192 
193                 /**
194                  * This event is triggered when sorting starts (belongs to jQuery UI).
195                  * @param event {Object} jQuery UI object.
196                  * @param ui {Object} jQuery UI object.
197                  **/
198                     start : function(event, ui) {
199                         /** When the sorting starts make sure the placeholder resembles the sorted item. */
200                         ui.placeholder.text(ui.helper.text());
201 
202                     },
203 
204                 /**
205                  * This event is triggered when a sortable item has been dragged out
206                  * from the list and into another (belongs to jQuery UI).
207                  * @param event {Object} jQuery UI object.
208                  * @param ui {Object} jQuery UI object.
209                  */
210                     remove : function(event, ui) {
211                         /** Id of the item being sorted. */
212                         var dimension_id = ui.item.find('a').attr('rel').split('_')[0];
213                         /** Remove the not-draggable class and add ui-draggable to all the sorted items sibilings. */
214                         $dimension_tree.find('[rel=' + dimension_id + ']').parent().children().children()
215                         .removeClass('not-draggable').addClass('ui-draggable');
216                     },
217 
218                 /**
219                  * This event is triggered when sorting stops, but when the placeholder/helper
220                  * is still available (belongs to jQuery UI).
221                  * @param event {Object} jQuery UI object.
222                  * @param ui {Object} jQuery UI object.
223                  */
224 
225                     beforeStop : function(event, ui) {
226                         /** If the item is not being 'dropped/removed' onto the .sidebar. */
227                         if(!(ui.item.hasClass('dropped'))) {
228                             /** Id of the removed item. */
229                             var dimension_id = ui.item.find('a').attr('rel').split('_')[0];
230                             /** Add the not-draggable class and add ui-draggable to all the sorted items sibilings. */
231                             $dimension_tree.find('[rel=' + dimension_id + ']').parent().children().children()
232                             .removeClass('ui-draggable').addClass('not-draggable');
233                             ui.item.addClass('dropped_dimension');
234                         }
235                     },
236                     
237                     /**
238                      * This event is triggered when sorting has stopped (belongs to jQuery UI).
239                      * @param event {Object} jQuery UI object.
240                      * @param ui {Object} jQuery UI object.
241                      */
242                     stop : function(event, ui) {
243                         /** Check if the placeholder should be visible. */
244                         if ($row_dropzone.find('.dropped_dimension').length == 0) {
245                             $row_dropzone.find('.placeholder_dimension').show();
246                         }else{
247                             $row_dropzone.find('.placeholder_dimension').hide();
248                         }
249                         if ($column_dropzone.find('.dropped_dimension').length == 0){
250                             $column_dropzone.find('.placeholder_dimension').show();
251                         }else{
252                             $column_dropzone.find('.placeholder_dimension').hide();
253                         }
254                     }
255                 }).disableSelection();
256 
257                 /** Measures. */
258                 
259                 /** Enable dragging on the dimension tree. */
260                 $measure_tree.draggable({
261                     cancel : ".not-draggable",
262                     helper : "clone",
263                     opacity : 0.60,
264                     tolerance : "pointer",
265                     /** Connect to the dimension dropzone. */
266                     connectToSortable : $measures_dropzone,
267                     cursorAt : {
268                         top : 7,
269                         left : 5
270                     }
271                 }).disableSelection();
272 
273 
274                 /** Make the dimension lists sortable and connectable. */
275                 $measures_dropzone.sortable({
276                     cancel : '.placeholder_measure',
277                     placeholder : 'placeholder_measure_sort',
278                     items : 'li:not(.placeholder_measure)',
279                     connectWith : $measures_dropzone,
280                     tolerance : "pointer",
281                     opacity: 0.60,
282                     cursorAt : {
283                         top : 7,
284                         left : 5
285                     },
286 
287                 /**
288                  * This event is triggered when sorting starts (belongs to jQuery UI).
289                  * @param event {Object} jQuery UI object.
290                  * @param ui {Object} jQuery UI object.
291                  **/
292                     start : function(event, ui) {
293                         /** When the sorting starts make sure the placeholder resembles the sorted item. */
294                         ui.placeholder.text(ui.helper.text());
295                     },
296 
297                     /**
298                  * This event is triggered when a sortable item has been dragged out
299                  * from the list and into another (belongs to jQuery UI).
300                  * @param event {Object} jQuery UI object.
301                  * @param ui {Object} jQuery UI object.
302                  */
303                     remove : function(event, ui) {
304                         /** Id of the item being sorted. */
305                         var measure_id = ui.item.find('a').attr('rel');
306                         /** Remove the not-draggable class and add ui-draggable to the measure item. */
307                         $measure_tree.find('[rel=' + measure_id + ']').parent()
308                         .removeClass('not-draggable').addClass('ui-draggable');
309                     },
310 
311                     /**
312                  * This event is triggered when sorting stops, but when the placeholder/helper
313                  * is still available (belongs to jQuery UI).
314                  * @param event {Object} jQuery UI object.
315                  * @param ui {Object} jQuery UI object.
316                  */
317 
318                     beforeStop : function(event, ui) {
319                         /** If the item is not being 'dropped/removed' onto the .sidebar. */
320                         if(!(ui.item.hasClass('dropped'))) {
321                             /** Id of the measure item. */
322                             var measure_id = ui.item.find('a').attr('rel');
323                             /** If this is the first measure being dropped. */
324                             if ($measures_dropzone.find('.dropped_measure').length == 0) {
325                                 /** Add the not-draggable class and remove ui-draggable to the measure item in the measures tree. */
326                                 $measure_tree.find('[rel=' + measure_id + ']').parent()
327                                 .removeClass('ui-draggable').addClass('not-draggable');
328                                 // Style the measure item.
329                                 ui.item.addClass('dropped_measure');
330                             }else{
331                                 /** Setup sorting from pointers. */
332                                 var drag_to = ui.helper.parent().parent().attr('class').split(' ')[1],
333                                 drag_from = $(this).parent().attr('class').split(' ')[1];
334                                 /** If sorting between lists only. */
335                                 if (drag_to === 'rows' && drag_from === 'columns' || drag_to === 'columns' && drag_from === 'rows') {
336                                     /** If sorting to an axis which has no measures then bring along all other measures. */
337                                     if ($('.'+drag_to).find('.dropped_measure').length - 1 == 0) {
338                                         /** Find all other measures and append them to the new axis. */
339                                         $measures_dropzone.find('.dropped_measure').appendTo($('.'+drag_to+' .measures_group'));
340                                     }
341                                 }else{
342                                     /** If sorting to a list which contains no measures. */
343                                     if ($('.'+drag_from).find('.dropped_measure').length == 0) {
344                                         /** Add the not-draggable class and remove ui-draggable to the measure item in the measures tree. */
345                                         $measure_tree.find('[rel=' + measure_id + ']').parent()
346                                         .removeClass('ui-draggable').addClass('not-draggable');
347                                         /** Find all other measures and append them to the new axis. */
348                                         $measures_dropzone.find('.dropped_measure').appendTo($('.'+drag_from+' .measures_group'));
349                                         // Style the measure item.
350                                         ui.item.addClass('dropped_measure');
351                                     }else{
352                                         /** If sorting from the tree to a list which already contains measures. */
353                                         $measure_tree.find('[rel=' + measure_id + ']')
354                                         .parent().removeClass('ui-draggable').addClass('not-draggable');
355                                         // Style the measure item.
356                                         ui.item.addClass('dropped_measure');
357                                     }
358                                 }
359                             }
360                         }
361                     },
362 
363                     /**
364                      * This event is triggered when sorting has stopped (belongs to jQuery UI).
365                      * @param event {Object} jQuery UI object.
366                      * @param ui {Object} jQuery UI object.
367                      */
368                     stop : function(event, ui) {
369                         /** Check if the placeholder should be visible. */
370                         if ($row_dropzone.find('.dropped_measure').length == 0) {
371                             $row_dropzone.find('.placeholder_measure').show();
372                         }else{
373                             $row_dropzone.find('.placeholder_measure').hide();
374                         }
375                         if ($column_dropzone.find('.dropped_measure').length == 0){
376                             $column_dropzone.find('.placeholder_measure').show();
377                         }else{
378                             $column_dropzone.find('.placeholder_measure').hide();
379                         }
380                     }
381                 }).disableSelection();
382 
383                 /**
384                  * Make the sidebar a droppable area for items.
385                  */
386                 $('.sidebar').droppable({
387                     accept : ".rows li, .columns li",
388 
389                     /**
390                      * This event is triggered when an accepted draggable is dropped 'over'
391                      * (within the tolerance of) this droppable (this belongs to jQuery UI).
392                      * @param event {Object} jQuery UI object.
393                      * @param ui {Object} jQuery UI object.
394                      */
395                     drop : function(event, ui) {
396                         /** Add a class to the to be dropped item so that the sortable beforeStop event can detect the item being removed. */
397                         ui.draggable.addClass('dropped');
398                         /** Remove the draggable item. */
399                         ui.draggable.remove();
400                         /** Using setTimeout due to IE7+ bug with jQuery UI. */
401                         setTimeout(function() {
402                             ui.draggable.remove();
403                         } , 1);
404                     }
405                 });
406             },
407             error: function() {
408                 view.show_dialog("Error", "Couldn't create a new query. Please try again.", "error");
409             }
410         });
411     }
