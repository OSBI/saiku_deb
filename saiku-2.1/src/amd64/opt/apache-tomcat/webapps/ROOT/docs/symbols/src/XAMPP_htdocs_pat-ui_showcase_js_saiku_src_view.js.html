 10 /** Resize the height of the layout to keep consistant columns. */
 11 function resize_height() {
 12     // Get the browser's current height
 13     var window_height = $(window).height();
 14     // When the height of the browser is less than 600px set a height of 600px.
 15     if(window_height <= 600) {
 16         window_height = 600;
 17     }
 18     var workspace_offset = $('.workspace_inner').outerHeight(true) - $('.workspace_inner').height(),
 19     // Add 1px to tabs height for tab_panel border-top: 1px solid #CCC
 20     sidebar_offset = ($('#toolbar').outerHeight(true) + ($('#tabs').outerHeight(true) + 1)),
 21     sidebar_height = window_height - sidebar_offset,
 22     workspace_height = sidebar_height - workspace_offset;
 23 
 24     $('.sidebar, .sidebar_separator').css('height', sidebar_height);
 25     $('.workspace_inner').css('height', workspace_height);
 26 }
 27 
 28 /** Toggle (hide/show) the sidebar. */
 29 function toggle_sidebar() {
 30     // Get the width of the sidebar.
 31     var sidebar_width = $('.sidebar').width();
 32     // If the sidebar is not hidden.
 33     if(sidebar_width == 260) {
 34         $('.sidebar').css('width', 0);
 35         $('.workspace_inner').css('margin-left', 5);
 36     } else {
 37         // If the sidebar is hidden.
 38         $('.sidebar').css('width', 260);
 39         $('.workspace_inner').css('margin-left', 265);
 40     }
 41 }

 43 /**
 44  * View class.
 45  * @class
 46  */
 47 var view = {
 48 
 49     /** Display the login form when the view is initialised. */
 50     init : function() {
 51         // Append a dialog <div/> to the body.
 52         $('<div id="dialog" class="dialog hide" />').appendTo('body');
 53         // Load the view into the dialog <div/> and disable caching.
 54         $.ajax({
 55             url : '../views/session',
 56             cache : false,
 57             dataType : "html",
 58             success : function(data) {
 59                 $('#dialog').html(data).modal({
 60                     overlayCss: {
 61                         background: 'white',
 62                         opacity: 'none'
 63                     }
 64                 });
 65                 $('#dialog').find('#login').click(function(){
 66                     model.username = $('#username').val();
 67                     model.password = $('#password').val();
 68                     model.get_session();
 69                 });
 70             }
 71         });
 72     },
 73 
 74     /** Initialise the user interface. */
 75     draw_ui : function () {
 76         // Show waiting message.
 77         view.show_waiting('Drawing UI, please wait...');
 78         
 79         /** Show all UI elements. */
 80         $('#toolbar, #tabs, #tab_panel').show();
 81 
 82         /** Add an event handler to all toolbar buttons. */
 83         $("#toolbar ul li a").click(function() {
 84             controller.toolbar_click_handler($(this));
 85             return false;
 86         });
 87 
 88         /** Bind resize_height() to the resize event. */
 89         $(window).bind('resize', function() {
 90             resize_height();
 91         });
 92 
 93         /** Bind toggle_sidebar() to click event on the sidebar_separator. */
 94         $('.sidebar_separator').bind('click', function() {
 95             toggle_sidebar();
 96         });
 97 
 98         /** Initialise resize_height() on first page load. */
 99         resize_height();
100 
101         // Remove waiting message.
102         view.stop_waiting();
103         
104     },
105 
106     /** Destroy the user interface. */
107     destroy_ui : function () {
108         $('#toolbar, #tabs, #tab_panel').hide();
109     },
110 
111     /**
112      * Displays a waiting dialog box.
113      * @param message {String} Waiting message to be displayed.
114      */
115     show_waiting : function (message) {
116         // Append the waiting <div/> to the body.
117         $('<div id="waiting" class="waiting hide" />').appendTo('body');
118         // Load the contents and message into the waiting <div/>.
119         $('#waiting').append('<div class="waiting_inner"><div class="waiting_body">' + message + '</div></div>').modal();
120     },
121     
122     /** Removes a waiting dialog box. */
123     stop_waiting : function () {
124         $.modal.close();
125         $('#waiting').remove();
126     },
127 
128     /**
129      * Load views into a dialog template
130      * @param url {String} The url where the view is located.
131      */
132     show_dialog : function(url) {
133         // Append a dialog <div/> to the body.
134         $('<div id="dialog" class="dialog hide" />').appendTo('body');
135         // Load the view into the dialog <div/> and disable caching.
136         $.ajax({
137             url : url,
138             cache : false,
139             dataType : "html",
140             success : function(data) {
141                 $('#dialog').html(data).modal();
142             }
143         });
144     }
145 
146 }
147 
148 /** Initialise the user interface. */
149 view.init();
