1 /**
  2  * @fileOverview    This represents the model for Saiku UI.
  3  * @description     This will handle all interaction to Saiku's REST API.
  4  * @version         1.0.0
  5  */
  6 
  7 /**
  8  * Model class
  9  * @class
 10  */
 11 var model = {
 12 
 13     /** Username to be used with BASIC_AUTH. */
 14     username : "",
 15 
 16     /** Password to be used with BASIC_AUTH. */
 17     password : "",
 18     
 19     /**
 20      * Handle all AJAX requests.
 21      * @param paramters {Object} Parameters for AJAX requests.
 22      */
 23     request : function(parameters) {
 24         if (typeof parameters.method == "undefined")
 25             parameters.method = "GET";
 26         if (typeof parameters.data == "undefined")
 27             parameters.data = {};
 28         if (typeof parameters.success == "undefined")
 29             parameters.success = function() {};
 30         if (typeof parameters.error == "undefined")
 31             parameters.error = model.server_error;
 32         $.ajax({
 33             type: parameters.method,
 34             url: "/pat-ui/fixtures/rest/" + parameters.url,
 35             //url: "rest/" + parameters.url, CHANGE
 36             dataType: 'json',
 37             username: model.username,
 38             password: model.password,
 39             success: parameters.success,
 40             error: parameters.error
 41         });
 42     },
 43     
 44     /** Get the sessionid and based on the username and unhide the UI. */
 45     get_session : function() {
 46         model.request({
 47             method: "POST",
 48             url: model.username + "/session",
 49             success: function(data, textStatus, XMLHttpRequest) {
 50                 model.session_id = data['@sessionid'];
 51                 model.connections = data;
 52                 view.draw_ui();
 53             }
 54         });
 55     }
 56     
 57 }
