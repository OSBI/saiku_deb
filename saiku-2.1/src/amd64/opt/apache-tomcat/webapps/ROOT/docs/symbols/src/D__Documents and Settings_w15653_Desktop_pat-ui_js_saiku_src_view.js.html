 10 /** Resize the height of the layout to keep consistant columns. */
 11 function resize_height() {
 12     // Get the browser's current height
 13     var window_height = $(window).height();
 14     // When the height of the browser is less than 600px set a height of 600px.
 15     if (window_height <= 600) {
 16         window_height = 600;
 17     }
 18     // Add 1px to tabs height for tab_panel border-top: 1px solid #CCC
 19     var sidebar_offset = ( $('#toolbar').outerHeight(true) + ($('#tabs').outerHeight(true) + 4) ),
 20     sidebar_height = window_height - sidebar_offset,
 21     workspace_height = sidebar_height - 20;
 22 
 23     $('.sidebar, .sidebar_separator').css('height', sidebar_height);
 24     $('.workspace_inner').css('height', workspace_height);
 25 }

 27 /** Toggle (hide/show) the sidebar. */
 28 function toggle_sidebar() {
 29     // TODO - Has to be for selected tab only.
 30     // Get the width of the sidebar.
 31     var sidebar_width = $('.sidebar').width();
 32     // If the sidebar is not hidden.
 33     if (sidebar_width == 260) {
 34         $('.sidebar').css('width', 0);
 35         $('.workspace_inner').css('margin-left', 5);
 36     } else {
 37         // If the sidebar is hidden.
 38         $('.sidebar').css('width', 260);
 39         $('.workspace_inner').css('margin-left', 265);
 40     }
 41 }

 53 /**
 54  * Container object used for tracking tabs.
 55  * @class
 56  */
 57 var TabContainer = function(tab_container, content_container) {
 58     $("<ul />").appendTo(tab_container);
 59     this.tab_container = tab_container.find("ul");
 60     this.content_container = content_container;
 61     this.tabs = new Array();
 62     this.selected = 0;
 63 
 64     /** Counts active tabs. */
 65     this.active_tabs = function() {
 66         active_tabs = 0;
 67         for (i = 0; i < this.tabs.length; i++) {
 68             if (typeof this.tabs[i] != "undefined") {
 69                 active_tabs++;
 70             }
 71         }
 72         return active_tabs;
 73     }
 74 
 75     /** Remove a tab and reclaim memory. */
 76     this.remove_tab = function(index) {
 77         if (typeof this.tabs[index] != "undefined") {
 78             this.tabs[index].tab.remove();
 79             this.tabs[index].content.remove();
 80             delete this.tabs[index].data;
 81             delete this.tabs[index];
 82         }
 83         // Find the next tab and select it.
 84         for (next_tab = index; next_tab < this.tabs.length; next_tab++) {
 85             if (typeof this.tabs[next_tab] != "undefined") {
 86                 this.select_tab(next_tab);
 87                 return;
 88             }
 89         }
 90         // Check and make sure there are any tabs at all.
 91         if (this.active_tabs() == 0 && model.session_id != "") {
 92             // Create one if not.
 93             this.add_tab();
 94             return;
 95         }
 96         // If the last tab was removed, select the next to last tab.
 97         this.select_tab(this.index_from_tab(this.tab_container.find("li:last")));
 98     };
 99 
100     /** Change the selected tab. */
101     this.select_tab = function(index) {
102         if (typeof this.tabs[index] != "undefined") {
103             this.tab_container.find("li.selected").removeClass("selected");
104             this.content_container.find(".tab").hide();
105             this.tabs[index].tab.addClass("selected");
106             resize_height();
107             this.tabs[index].content.show();
108             this.selected = index;
109         }
110     };
111 
112     /** Add a tab. */
113     this.add_tab = function() {
114         // Create the tab itself
115         var new_index = this.tabs.length;
116         var $new_tab = $("<li />");
117         var $new_tab_link = $("<a />")
118         .html("Unsaved query (" + (new_index + 1) + ")")
119         .appendTo($new_tab);
120         var $new_tab_closer = $("<span>Close Tab</span>")
121         .addClass("close_tab")
122         .appendTo($new_tab);
123         $new_tab.appendTo(this.tab_container);
124 
125         // Create the content portion of the tab.
126         $new_tab_content = $('<div />')
127         .addClass("tab")
128         .load("views/queries/", function() {
129             view.load_cubes(new_index);
130             resize_height();
131         });
132         $new_tab_content.appendTo(this.content_container);
133 
134         // Register the new tab with the TabContainer.
135         this.tabs.push(new Tab($new_tab, $new_tab_content));
136         this.select_tab(new_index);
137 
138     };
139 
140     /** Empty the tab container (used for logout) */
141     this.clear_tabs = function() {
142         for (i = 0; i < this.tabs.length; i++) {
143             if (typeof this.tabs[i] != 'undefined') {
144                 this.remove_tab(i);
145             }
146         }
147     };
148 
149     /** Determine whether or not the TabContainer is empty .*/
150     this.is_empty = function() {
151         // If the array is uninitialized, obviously return true
152         if (this.tabs.length == 0)
153             return true;
154         // Check to see if there are any active tabs
155         for (i = 0; i < this.tabs.length; i++) {
156             if (typeof this.tabs[i] != 'undefined') {
157                 // An active tab still exists, return false
158                 return false;
159             }
160         }
161         // If not, return true
162         return true;
163     };
164 
165     /** Get a tab_index from a tab instance. */
166     this.index_from_tab = function($tab) {
167         for (i = 0; i < this.tabs.length; i++) {
168             if (typeof this.tabs[i] != "undefined" && $tab[0] == this.tabs[i].tab[0]) {
169                 return i;
170             }
171         }
172         return -1;
173     };
174 
175     /** Get a tab_index from a tab content. */
176     this.index_from_content = function($content) {
177         for (i = 0; i < this.tabs.length; i++) {
178             if (typeof this.tabs[i] != "undefined" && $content[0] == this.tabs[i].content[0]) {
179                 return i;
180             }
181         }
182         return -1;
183     };
184 };

283     /**
284      * Populate a select box with available schemas and cubes.
285      * @param tab_index {Integer} Index of the selected tab.
286      */
287     load_cubes : function(tab_index) {
288         view.start_waiting('Loading available schemas and cubes');
289         $tab = view.tabs.tabs[tab_index].content;
290         $cubes = $tab.find('.cubes');
291         $cubes.append('<option>Select a cube</option>');
292 
293         view.tabs.tabs[tab_index].data['navigation'] = new Array();
294         storage_id = 0;
295 
296         /** Loop through available connections and populate the select box. */
297         $.each(model.connections.connections.connection, function(i,connection){
298             $.each(connection.schemas, function(i,schema){
299                 
300                 $cubes.append('<optgroup label="'+schema['@schemaname']+'">');
301                 $.each(schema.cubes, function(i,cube){
302                     if (cube.length == undefined)
303                         cube = [cube];
304                     $.each(cube, function(i,item){
305                         $("<option />")
306                         .attr({
307                             'value': storage_id
308                         })
309                         .text(item['@cubename'])
310                         .appendTo($cubes);
311                         view.tabs.tabs[tab_index].data['navigation'][storage_id] = {
312                             'connection_id': connection['@connectionid'],
313                             'schema': schema['@schemaname'],
314                             'cube': item['@cubename']
315                         };
316                         storage_id++;
317                     });
318                 });
319                 $cubes.append('</optgroup>');
320             });
321         });
322         view.stop_waiting();
323 
324         $cubes.change(function() {
325             model.new_query(tab_index);
326         });
327     },

329     /**
330      * Populate the dimension tree for the selected tab.
331      * @param $tab {Object} Selected tab content.
332      * @param data {Object} Data object which contains the available dimension
333      *                      members.
334      */
335     load_dimensions : function($tab, data) {
336         // Remove any instances of a tree.
337         $tab.find('.dimension_tree ul').remove();
338         // Add a new dimension tree.
339         $dimension_tree = $('<ul />').appendTo($tab.find('.dimension_tree')).addClass('dtree');
340         // Populate the tree with first level dimensions.
341         $.each(data, function(i, dimension) {
342             if (this['@dimensionname'] != 'Measures') {
343                 // Make sure the first level has a unique rel attribute.
344                 $first_level = $('<li><a href="#" rel="d' + i + '" class="folder_collapsed root">' + this['@dimensionname'] + '</a></li>')
345                 .addClass("collapsed")
346                 .appendTo($dimension_tree);
347                 if (dimension.levels.length > 1) {
348                     $second_level = $('<ul />').appendTo($first_level);
349                     $.each(dimension.levels, function(j, level){
350                         $li = $('<li />').mousedown(function() {
351                             return false;
352                         }).appendTo($second_level);
353                         // Create a parent-child relationship with the rel attribute.
354                         $second_level_link = $('<a href="#" class="dimension" rel="d' + i + '_' + j + '" title="' + this['@levelname'] + '">' + level['@levelcaption'] + '</a>')
355                         .appendTo($li);
356                     });
357                 }
358             }
359         });
360     },
