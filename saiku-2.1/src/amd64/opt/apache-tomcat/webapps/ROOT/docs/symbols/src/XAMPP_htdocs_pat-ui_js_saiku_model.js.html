 27     /**
 28      * Check if the session id is valid and display the login form.
 29      * @param username {String} The username to use with Saiku server.
 30      */
 31     init: function (username) {
 32         // Let's be friends!
 33         model.server_errors = 0;
 34 
 35         /**
 36          * Ask for credentials using a pretty form that doesn't block the 
 37          * browser (and hides password)
 38          */   
 39         if (model.username == "" || model.password == "") {
 40             $('#login_form').jqm({
 41                 modal: true
 42             });
 43             $('#login_form').jqmShow()
 44             .find('#login')         
 45             .click(function(){
 46                 $('#login_form').jqmHide();
 47                 model.username = $('#login_form .username').val();
 48                 model.password = $('#login_form .password').val();
 49                 model.get_session();
 50             });
 51         }
 52     },
 53     
 54     /** Handle all AJAX requests. */
 55     request: function(parameters) {
 56         if (typeof parameters.method == "undefined")
 57             parameters.method = "GET";
 58         if (typeof parameters.data == "undefined")
 59             parameters.data = {};
 60         if (typeof parameters.success == "undefined")
 61             parameters.success = function() {};
 62         if (typeof parameters.error == "undefined")
 63             parameters.error = controller.server_error;
 64     	
 65         $.ajax({
 66             type: parameters.method,
 67             //url: "/pat-ui/fixtures/rest/" + parameters.url,
 68             url: "rest/" + parameters.url,
 69             dataType: 'json',
 70             username: model.username,
 71             password: model.password,
 72             success: parameters.success,
 73             error: parameters.error
 74         });
 75     },
 76 	
 77     /** Obtain a session and load connections. */
 78     get_session: function() {
 79         model.request({
 80             method: "POST",
 81             url: model.username + "/session",
 82             success: function(data, textStatus, XMLHttpRequest) {
 83                 model.session_id = data['@sessionid'];
 84                 model.connections = data;
 85                 view.new_tab();
 86                 view.login();
 87             }
 88         });
 89     },

100     /**
101      * Populate dimension and measure trees with available items, enable
102      * draggable, droppable and sortable lists.
103      */
104     new_query: function(tab_index) {
105 
106         // Patch for Chrome/Safari showing the text cursor when dragging
107         document.onselectstart = function () {
108             return false;
109         };
110 
111         // Find the selected cube
112         $cube = view.tabs.tabs[tab_index].content.find(".cubes option:selected");
113     	
114         data = view.tabs.tabs[tab_index].data['navigation'][$cube.attr('value')];
115         if (typeof data == "undefined") {
116             debug("Broken tab: " + tab_index + ", cube value: " + $cube.attr('value'));
117             //view.tabs.tabs[tab_index].content.html("Something broke. Please close this tab and try again on another one.");
118             view.info("Please close this tab and try again on another one.","Error");
119             return;
120         }
121 
122         view.processing("Creating new query on " + data['cube']);
123         $tab = view.tabs.tabs[tab_index].content;
124     	
125         model.request({
126             method: "POST",
127             url: model.username + "/query/" + data['schema'] + "/" + data['cube'] + "/newquery",
128             success: function(data, textStatus, XMLHttpRequest) {
129                 // Remove all instances of previous trees
130                 // in case user reselects cube
131                 $tab.find('.dimensions_tree ul, .measures_tree ul').remove();
132                 // FIXME - this needs to be a recursive algorithm in the view
133         	
134                 $dimension_tree = $('<ul />').appendTo($tab.find('.dimensions_tree'));
135                 $measures_tree = $('<ul />').appendTo($tab.find('.measures_tree'));
136                 
137                 $both_trees = $tab.find('.dimensions_tree, .measures_tree');
138 
139                 $.each(data.axis.dimensions, function(i,dimension) {
140 
141                     // Populate dimension tree first
142                     if(this['@dimensionname'] != 'Measures') {                        
143                         // Append <li/> to the dimensions_tree <ul/>
144                         $first_level = $('<li><a href="#">'+this['@dimensionname']+'</a></li>')
145                         .addClass("root")
146                         .appendTo($dimension_tree);
147 
148                         if (dimension.levels.length > 1) {
149                             // Append <ul/> to the dimensions_tree <li/>
150                             $second_level = $('<ul />').appendTo($first_level);
151 	                        
152                             // If there is a secondary level loop through and add to the <ul/>
153                             $.each(dimension.levels, function(j,level){
154                                 // Populate the second <ul/>
155                                 $li = $('<li />')
156                                 .mousedown(function() {
157                                     return false;
158                                 })
159                                 .appendTo($second_level);
160                                 
161                                 $second_level_link = $('<a href="#" class="dimension">'+level['@levelcaption']+'</a>')
162                                 .appendTo($li);
163                                 
164                                 if (level.members.length > 1) {
165                                 // PRASHANT - Commented this out
166                                 /*$li.addClass("folder");
167                                     $ul = $("<ul />").appendTo($li);
168                                     $.each(level.members, function(k, member) {
169                                         $li2 = $('<li />')
170                                         .mousedown(function() {
171                                             return false;
172                                         })
173                                         .appendTo($ul);
174                                 		 
175                                         $("<a href='#'>"+member['@membercaption']+"</a>")
176                                         .click(function(event) {
177                                             return false;
178                                         })
179                                         .appendTo($li2);
180                                     });*/
181                                 } else {
182                                     $second_level_link.click(function(event) {
183                                         return false;
184                                     });
185                                 }                                
186                             });
187                         }
188                         
189                     } else {
190                         // Create a 'dummy' measures <ul/>
191                         // Append <li/> to the dimensions_tree <ul/>
192                         $measures = $('<li><a href="#">Measures</a></li>')
193                         .addClass("root")
194                         .appendTo($tab.find('.measures_tree ul'));
195 
196                         // Append <ul/> to the dimensions_tree <li/>
197                         $measures_ul = $('<ul />').appendTo($measures);
198                             
199                         $.each(dimension.levels.members, function(i,member){
200                             // Populate the second <ul/>
201                             $('<li id="'+this['@membercaption']+'"><a href="#" class="measure">'+this['@membercaption']+'</a></li>')
202                             .mousedown(function() {
203                                 return false;
204                             })
205                             .appendTo($measures_ul);
206                         });
207                         
208                     }
209                 });
210                 
211                 // Hide all children
212                 $tab.find(".root").children("ul").children("li").hide();
213 
214                 $tab.find(".root").click(function() {
215                     $(this).children("ul").children("li").toggle();
216 
217                     if($(this).hasClass("expand")) {
218                         $(this).removeClass("expand");
219                     }else{
220                         $(this).addClass("expand");
221                     }
222 
223                     return false;
224                 });
225         
226                 /* DND Rules:
227                  * Rules:
228                  * Unique items only the row and column axis
229                  * You can only drag one item from the hierarchy onto 
230                  * 		the row and column axis i.e. Only Region from Markets
231                  * Once the above occurs you can no longer any other items 
232                  * 		from the hierarchy onto the row, column, filters axis
233                  * Measures must be grouped together and can only belong 
234                  * 		on one axis as a group
235                  * Measures can not be place on the filters axis
236                  */
237 
238                 //  Draggable
239                 $both_trees.find("li ul li a").draggable({
240                     cancel: ".not-draggable .placeholder",
241                     helper: "clone",
242                     opacity: 0.90,
243                     start: function(event, ui) {
244                         if(ui.helper.hasClass('dimension')) {
245                             ui.helper.addClass('dimension_item');
246                         }else if(ui.helper.hasClass('measure')) {
247                             ui.helper.addClass('measure_item');
248                         }
249                     }
250                 });
251 
252                 $drop_zones = view.tabs.tabs[tab_index].content.find('.dropzone_rows ul, .dropzone_columns ul, .dropzone_filters ul');
253 
254                 //  Droppable
255                 $drop_zones.droppable({
256                     accept: '.ui-draggable',
257                     activeClass: "notice",
258                     hoverClass: "success",
259                     drop: function(event, ui) {
260 
261                         $(this).find('.placeholder').remove();
262 
263                         if(ui.helper.hasClass('dimension')) {
264                             $(this).append('<li class="dimension_item"><span>'+ui.draggable.text()+'</span></li>');
265                         }else if(ui.helper.hasClass('measure')) {
266                             $(this).append('<li class="measure_item"><span>'+ui.draggable.text()+'</span></li>');
267                         }
268 
269                         tab_index = view.tabs.index_from_content($(this).parent().parent().parent().parent().parent("div.tab"));
270                         view.check_query(tab_index);
271                     }
272                 }).sortable({
273                     cancel: 'placeholder',
274                     connectWith: $drop_zones,
275                     tolerance: 'pointer',
276                     placeholder: 'placeholder_sort',
277                     forcePlaceholderSize: true
278                 });
279                 
280                 $('li', $drop_zones).draggable({
281                     cancel: ".not-draggable",
282                     revert: "invalid",
283                     opacity: 0.90
284                 });
285 
286                 $('.sidebar').droppable({
287                     accept: ".dropzone_rows ul li, .dropzone_columns ul li, .dropzone_filters ul li",
288                     drop: function(event, ui) {
289                         // Using setTimeout due to IE7+ bug with jQuery UI
290                         // Info: http://dev.jqueryui.com/ticket/4550
291                         setTimeout(function() {
292                             ui.draggable.remove();
293                         },1);
294 
295                         // FIXME - Incorrect logic to keeping placeholder 
296                         if(!$(".dropzone_rows ul li").hasClass("placeholder")) {
297                             $(".dropzone_rows ul").append('<li class="placeholder">Drop fields here</li>');
298                         }else if(!$(".dropzone_columns ul li").hasClass("placeholder")) {
299                             $(".dropzone_columns ul").append('<li class="placeholder">Drop fields here</li>');
300                         }else if(!$(".dropzone_filters ul li").hasClass("placeholder")) {
301                             $(".dropzone_filters ul").append('<li class="placeholder">Drop fields here</li>');
302                         }
303                     }
304                 })
305                 
306                 //.selectable();
307                 // can use selectable to select and then delete using the "delete or backspace key"
308 
309                 // Remove blockUI
310                 view.free();
311             },
312             error: function() {
313                 view.info("Couldn't create a new query. Please try again.", "Error");
314                 view.free();
315             }
316         });
317     },

328     /** Kill credentials and server side session. */
329     logout: function() {
330         // Kill server-side session
331         model.request({
332             url: "",
333             success: function() {
334                 // Clear credentials
335                 model.username = "";
336                 model.password = "";
337                 model.session_id = "";
338                 model.connections = {};
339 
340                 // Remove all tabs
341                 view.tabs.clear_tabs();
342 
343                 // Hide everything
344                 view.logout();
345                 
346                 // Refresh the page
347                 location.reload(true);
348             }
349         });
350     },
